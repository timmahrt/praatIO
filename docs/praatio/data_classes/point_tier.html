 19def _homogenizeEntries(entries):
 20    """
 21    Enforces consistency in points
 22
 23    - converts all entries to points
 24    - removes whitespace in labels
 25    - sorts values by time
 26    """
 27    processedEntries = [Point(float(time), label.strip()) for time, label in entries]
 28    processedEntries.sort()
 29    return processedEntries

 32def _calculateMinAndMaxTime(entries: Sequence[Point], minT=None, maxT=None):
 33    timeList = [time for time, label in entries]
 34    if minT is not None:
 35        timeList.append(float(minT))
 36    if maxT is not None:
 37        timeList.append(float(maxT))
 38
 39    try:
 40        calculatedMinT = min(timeList)
 41        calculatedMaxT = max(timeList)
 42    except ValueError:
 43        raise errors.TimelessTextgridTierException()
 44
 45    return (calculatedMinT, calculatedMaxT)

 53    def __init__(
 54        self,
 55        name: str,
 56        entries: List[Point],
 57        minT: Optional[float] = None,
 58        maxT: Optional[float] = None,
 59    ):
 60        """A point tier is for annotating instaneous events
 61
 62        The entries is of the form:
 63        [(timeVal1, label1), (timeVal2, label2), ]
 64
 65        The data stored in the labels can be anything but will
 66        be interpreted as text by praatio (the label could be descriptive
 67        text e.g. ('peak point here') or numerical data e.g. (pitch values
 68        like '132'))
 69        """
 70        entries = _homogenizeEntries(entries)
 71        calculatedMinT, calculatedMaxT = _calculateMinAndMaxTime(entries, minT, maxT)
 72
 73        super(PointTier, self).__init__(name, entries, calculatedMinT, calculatedMaxT)

 75    def crop(
 76        self,
 77        cropStart: float,
 78        cropEnd: float,
 79        mode: Literal["strict", "lax", "truncated"] = "lax",
 80        rebaseToZero: bool = True,
 81    ) -> "PointTier":
 82        """Creates a new tier containing all entries inside the new interval
 83
 84        Args:
 85            cropStart:
 86            cropEnd:
 87            mode: Mode is ignored.  This parameter is kept for
 88                compatibility with IntervalTier.crop()
 89            rebaseToZero: if True, all entries will have their
 90                timestamps subtracted by *cropStart*.
 91
 92        Returns:
 93            the modified version of the current tier
 94        """
 95        if cropStart >= cropEnd:
 96            raise errors.ArgumentError(
 97                f"Crop error: start time ({cropStart}) must occur before end time ({cropEnd})"
 98            )
 99
100        newEntries = []
101
102        for entry in self.entries:
103            timestamp = entry.time
104
105            if timestamp >= cropStart and timestamp <= cropEnd:
106                newEntries.append(entry)
107
108        if rebaseToZero is True:
109            newEntries = [
110                Point(timeV - cropStart, label) for timeV, label in newEntries
111            ]
112            minT = 0.0
113            maxT = cropEnd - cropStart
114        else:
115            minT = cropStart
116            maxT = cropEnd
117
118        return PointTier(self.name, newEntries, minT, maxT)

121        """Removes an entry from the entries"""

124    def editTimestamps(
125        self,
126        offset: float,
127        reportingMode: Literal["silence", "warning", "error"] = "warning",
128    ) -> "PointTier":
129        """Modifies all timestamps by a constant amount
130
131        Args:
132            offset:
133            reportingMode: one of "silence", "warning", or "error". This flag
134                determines the behavior if an entries moves outside of minTimestamp
135                or maxTimestamp after being edited
136
137        Returns:
138            the modified version of the current tier
139        """
140        utils.validateOption(
141            "reportingMode", reportingMode, constants.ErrorReportingMode
142        )
143        errorReporter = utils.getErrorReporter(reportingMode)
144
145        newEntries: List[Point] = []
146        for timestamp, label in self.entries:
147
148            newTimestamp = timestamp + offset
149            utils.checkIsUndershoot(newTimestamp, self.minTimestamp, errorReporter)
150            utils.checkIsOvershoot(newTimestamp, self.maxTimestamp, errorReporter)
151
152            if newTimestamp < 0:
153                continue
154
155            newEntries.append(Point(newTimestamp, label))
156
157        # Determine new min and max timestamps
158        timeList = [float(point.time) for point in newEntries]
159        newMin = min(timeList)
160        newMax = max(timeList)
161
162        if newMin > self.minTimestamp:
163            newMin = self.minTimestamp
164
165        if newMax < self.maxTimestamp:
166            newMax = self.maxTimestamp
167
168        return PointTier(self.name, newEntries, newMin, newMax)

170    def getValuesAtPoints(
171        self,
172        dataTupleList: List[Tuple[float, ...]],
173        fuzzyMatching: bool = False,
174    ) -> List[Tuple[Any, ...]]:
175        """Get the values that occur at points in the point tier
176
177        The procedure assumes that all data is ordered in time.
178        dataTupleList should be in the form
179        [(t1, v1a, v1b, ..), (t2, v2a, v2b, ..), ..]
180
181        It returns the data in the form of
182        [(t1, v1a, v1b, ..), (t2, v2a, v2b), ..]
183
184        The procedure makes one pass through dataTupleList and one
185        pass through self.entries.  If the data is not sequentially
186        ordered, the incorrect response will be returned.
187
188        Args:
189            dataTupleList:
190            fuzzyMatching: if True, if there is not a feature value
191                at a point, the nearest feature value will be taken.
192
193        Returns:
194            A list of values that exist at the given timepoints
195        """
196
197        currentIndex = 0
198        retList = []
199
200        sortedDataTupleList = sorted(dataTupleList)
201        for timestamp, label in self.entries:
202            retTuple = utils.getValueAtTime(
203                timestamp,
204                sortedDataTupleList,
205                fuzzyMatching=fuzzyMatching,
206                startI=currentIndex,
207            )
208            retRow, currentIndex = retTuple
209            retList.append(retRow)
210
211        return retList

213    def eraseRegion(
214        self,
215        start: float,
216        end: float,
217        collisionMode: Literal["truncate", "categorical", "error"] = "error",
218        doShrink: bool = True,
219    ) -> "PointTier":
220        """Makes a region in a tier blank (removes all contained entries)
221
222        Args:
223            start: the start of the deletion interval
224            end: the end of the deletion interval
225            collisionMode: Ignored for the moment (added for compatibility with
226                eraseRegion() for Interval Tiers)
227            doShrink: if True, moves leftward by (/end/ - /start/) all points
228                to the right of /end/
229
230        Returns:
231            The modified version of the current tier
232        """
233
234        newTier = self.new()
235        croppedTier = newTier.crop(start, end, constants.CropCollision.TRUNCATED, False)
236        matchList = croppedTier.entries
237
238        if len(matchList) > 0:
239            # Remove all the matches from the entries
240            # Go in reverse order because we're destructively altering
241            # the order of the list (messes up index order)
242            for point in matchList[::-1]:
243                newTier.deleteEntry(point)
244
245        if doShrink is True:
246            newEntries = []
247            diff = end - start
248            for point in newTier.entries:
249                if point.time < start:
250                    newEntries.append(point)
251                elif point.time > end:
252                    newEntries.append(Point(point.time - diff, point.label))
253
254            newMax = newTier.maxTimestamp - diff
255            newTier = newTier.new(entries=newEntries, maxTimestamp=newMax)
256
257        return newTier

259    def insertEntry(
260        self,
261        entry: Point,
262        collisionMode: Literal["replace", "merge", "error"] = "error",
263        collisionReportingMode: Literal["silence", "warning"] = "warning",
264    ) -> None:
265        """Inserts an interval into the tier
266
267        Args:
268            entry: the entry to insert
269            collisionMode: determines the behavior if intervals exist in
270                the insertion area.
271                - 'replace', existing items will be removed
272                - 'merge', inserting item will be fused with existing items
273                - 'error', will throw TextgridCollisionException
274            collisionReportingMode:
275
276        Returns:
277            None
278        """
279
280        utils.validateOption(
281            "collisionMode", collisionMode, constants.IntervalCollision
282        )
283        utils.validateOption(
284            "collisionReportingMode",
285            collisionReportingMode,
286            constants.ErrorReportingMode,
287        )
288        collisionReporter = utils.getErrorReporter(collisionReportingMode)
289
290        if not isinstance(entry, Point):
291            newPoint = Point(entry[0], entry[1])
292        else:
293            newPoint = entry
294
295        matchList = []
296        i = None
297        for i, point in enumerate(self.entries):
298            if point.time == newPoint.time:
299                matchList.append(point)
300                break
301
302        if len(matchList) == 0:
303            self._entries.append(newPoint)
304
305        elif collisionMode == constants.IntervalCollision.REPLACE:
306            self.deleteEntry(self.entries[i])
307            self._entries.append(newPoint)
308
309        elif collisionMode == constants.IntervalCollision.MERGE:
310            oldPoint = self.entries[i]
311            mergedPoint = Point(
312                newPoint.time, "-".join([oldPoint.label, newPoint.label])
313            )
314            self.deleteEntry(self._entries[i])
315            self._entries.append(mergedPoint)
316
317        else:
318            raise errors.CollisionError(
319                f"Attempted to insert interval {point} into tier {self.name} "
320                "of textgrid but overlapping entries "
321                f"{[tuple(interval) for interval in matchList]} "
322                "already exist"
323            )
324
325        self.sort()
326
327        if len(matchList) != 0:
328            collisionReporter(
329                errors.CollisionError,
330                f"Collision warning for ({point}) with items ({matchList}) of tier '{self.name}'",
331            )

365        self, reportingMode: Literal["silence", "warning", "error"] = "warning"
366    ) -> bool:
367        """Validate this tier
368
369        Returns whether the tier is valid or not. If reportingMode is "warning"
370        or "error" this will also print on error or stop execution, respectively.
371
372        Args:
373            reportingMode: Determines the behavior if there is a size difference
374                between the maxTimestamp in the tier and the current textgrid.
375
376        Returns:
377            True if this tier is valid; False otherwise
378        """
379        utils.validateOption(

382        errorReporter = utils.getErrorReporter(reportingMode)
383
384        isValid = True

386        for point in self.entries:

396                isValid = False
397
398            if utils.checkIsOvershoot(point.time, self.maxTimestamp, errorReporter):
399                isValid = False
400
401            previousPoint = point
402
403        return isValid
