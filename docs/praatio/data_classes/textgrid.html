 79    def addTier(
 80        self,
 81        tier: textgrid_tier.TextgridTier,
 82        tierIndex: Optional[int] = None,
 83        reportingMode: Literal["silence", "warning", "error"] = "warning",
 84    ) -> None:
 85        """Add a tier to this textgrid.
 86
 87        Args:
 88            tier: The tier to add to the textgrid
 89            tierIndex: Insert the tier into the specified position;
 90                if None, the tier will appear after all existing tiers
 91            reportingMode: This flag determines the behavior if there is a size
 92                difference between the maxTimestamp in the tier and the current
 93                textgrid.
 94
 95        Returns:
 96            None
 97
 98        Raises:
 99            TierNameExistsError: The textgrid already contains a tier with the same
100                name as the tier being added
101            TextgridStateAutoModified: The minimum or maximum timestamp was changed
102                when not permitted
103            IndexError: TierIndex is too large for the size of the existing tier list
104        """
105        utils.validateOption(
106            "reportingMode", reportingMode, constants.ErrorReportingMode
107        )
108        errorReporter = utils.getErrorReporter(reportingMode)
109
110        if tier.name in self.tierNames:
111            raise errors.TierNameExistsError("Tier name already in tier")
112
113        if tierIndex is None:
114            self._tierDict[tier.name] = tier
115        else:  # Need to recreate the tierDict with the new order
116            newOrderedTierNameList = list(self.tierNames)
117            newOrderedTierNameList.insert(tierIndex, tier.name)
118
119            newTierDict = OrderedDict()
120            self._tierDict[tier.name] = tier
121            for tmpName in newOrderedTierNameList:
122                newTierDict[tmpName] = self.getTier(tmpName)
123            self._tierDict = newTierDict
124
125        minV = tier.minTimestamp
126        if self.minTimestamp is not None and minV < self.minTimestamp:
127            errorReporter(
128                errors.TextgridStateAutoModified,
129                f"Minimum timestamp in Textgrid changed from ({self.minTimestamp}) to ({minV})",
130            )
131        if self.minTimestamp is None or minV < self.minTimestamp:
132            self.minTimestamp = minV
133
134        maxV = tier.maxTimestamp
135        if self.maxTimestamp is not None and maxV > self.maxTimestamp:
136            errorReporter(
137                errors.TextgridStateAutoModified,
138                f"Maximum timestamp in Textgrid changed from ({self.maxTimestamp}) to ({maxV})",
139            )
140        if self.maxTimestamp is None or maxV > self.maxTimestamp:

143    def appendTextgrid(self, tg: "Textgrid", onlyMatchingNames: bool) -> "Textgrid":
144        """Append one textgrid to the end of this one
145
146        Args:
147            tg: the textgrid to add to this one
148            onlyMatchingNames: if False, tiers that don't appear in both
149                textgrids will also appear
150
151        Returns:
152            the modified version of the current textgrid
153        """
154        minTime = self.minTimestamp
155        maxTime = self.maxTimestamp + tg.maxTimestamp
156        retTG = Textgrid(minTime, maxTime)
157
158        # Get all tier names.  Ordered first by this textgrid and
159        # then by the other textgrid.
160        combinedTierNames = list(self.tierNames)
161        for tierName in tg.tierNames:
162            if tierName not in combinedTierNames:
163                combinedTierNames.append(tierName)
164
165        # Determine the tier names that will be in the final textgrid
166        finalTierNames = []
167        if onlyMatchingNames is False:
168            finalTierNames = combinedTierNames
169        else:
170            for tierName in combinedTierNames:
171                if tierName in self.tierNames:
172                    if tierName in tg.tierNames:
173                        finalTierNames.append(tierName)
174
175        # Add tiers from this textgrid
176        for tierName in finalTierNames:
177            if tierName in self.tierNames:
178                tier = self.getTier(tierName)
179                retTG.addTier(tier)
180
181        # Add tiers from the given textgrid
182        for tierName in finalTierNames:
183            if tierName in tg.tierNames:
184                appendTier = tg.getTier(tierName)
185                appendTier = appendTier.new(minTimestamp=minTime, maxTimestamp=maxTime)
186
187                appendTier = appendTier.editTimestamps(self.maxTimestamp)
188
189                if tierName in retTG.tierNames:
190                    tier = retTG.getTier(tierName)
191                    newEntries = retTG.getTier(tierName).entries
192                    newEntries += appendTier.entries
193
194                    tier = tier.new(
195                        entries=newEntries,
196                        minTimestamp=minTime,
197                        maxTimestamp=maxTime,
198                    )
199                    retTG.replaceTier(tierName, tier)
200
201                else:
202                    tier = appendTier
203                    tier = tier.new(minTimestamp=minTime, maxTimestamp=maxTime)
204                    retTG.addTier(tier)
205
206        return retTG

208    def crop(
209        self,
210        cropStart: float,
211        cropEnd: float,
212        mode: Literal["strict", "lax", "truncated"],
213        rebaseToZero: bool,
214    ) -> "Textgrid":
215        """Creates a textgrid where all intervals fit within the crop region
216
217        Args:
218            cropStart: The start time of the crop interval
219            cropEnd: The stop time of the crop interval
220            mode: Determines the crop behavior
221                - 'strict', only intervals wholly contained by the crop
222                    interval will be kept
223                - 'lax', partially contained intervals will be kept
224                - 'truncated', partially contained intervals will be
225                    truncated to fit within the crop region.
226            rebaseToZero: if True, the cropped textgrid timestamps will be
227                subtracted by the cropStart; if False, timestamps will not
228                be changed
229
230        Returns:
231            the modified version of the current textgrid
232        """
233        utils.validateOption("mode", mode, CropCollision)
234
235        if cropStart >= cropEnd:
236            raise errors.ArgumentError(
237                f"Crop error: start time ({cropStart}) must occur before end time ({cropEnd})"
238            )
239
240        if rebaseToZero is True:
241            minT = 0.0
242            maxT = cropEnd - cropStart
243        else:
244            minT = cropStart
245            maxT = cropEnd
246        newTG = Textgrid(minT, maxT)
247
248        for tierName in self.tierNames:
249            tier = self.getTier(tierName)
250            newTier = tier.crop(cropStart, cropEnd, mode, rebaseToZero)
251
252            reportingMode: Literal[
253                "silence", "warning", "error"
254            ] = constants.ErrorReportingMode.WARNING
255            if mode == constants.CropCollision.LAX:
256                # We expect that there will be changes to the size
257                # of the textgrid when the mode is LAX
258                reportingMode = constants.ErrorReportingMode.SILENCE
259
260            newTG.addTier(newTier, reportingMode=reportingMode)
261
262        return newTG

264    def eraseRegion(self, start: float, end: float, doShrink: bool) -> "Textgrid":
265        """Makes a region in a tier blank (removes all contained entries)
266
267        Intervals that span the region to erase will be truncated.
268
269        Args:
270            start:
271            end:
272            doShrink: if True, all entries appearing after the
273                erased interval will be shifted to fill the void (ie
274                the duration of the textgrid will be reduced by
275                *start* - *end*)
276
277        Returns:
278            the modified version of the current textgrid
279
280        Raises:
281            ArgumentError
282        """
283        if start >= end:
284            raise errors.ArgumentError(
285                f"EraseRegion error: start time ({start}) must occur before end time ({end})"
286            )
287
288        diff = end - start
289
290        maxTimestamp = self.maxTimestamp
291        if doShrink is True:
292            maxTimestamp -= diff
293
294        newTG = Textgrid(self.minTimestamp, self.maxTimestamp)
295        for tier in self.tiers:
296            shrunkTier = tier.eraseRegion(
297                start, end, constants.EraseCollision.TRUNCATE, doShrink
298            )
299            newTG.addTier(shrunkTier)
300
301        newTG.maxTimestamp = maxTimestamp
302
303        return newTG

343    ) -> "Textgrid":

373            newTier = tier.insertSpace(start, duration, collisionMode)
374            newTG.addTier(newTier)

378    def mergeTiers(
379        self, tierNames: Optional[Sequence[str]] = None, preserveOtherTiers: bool = True
380    ) -> "Textgrid":
381        """Combine tiers
382
383        Args:
384            tierList: A list of tier names to combine. If none, combine
385                all tiers.
386            preserveOtherTiers: If false, uncombined tiers are not
387                included in the output.
388
389        Returns:
390            Textgrid: the modified version of the current textgrid
391        """
392        if tierNames is None:
393            tierNames = self.tierNames
394
395        # Determine the tiers to merge
396        intervalTierNames = []
397        pointTierNames = []
398        for tierName in tierNames:
399            tier = self.getTier(tierName)
400            if isinstance(tier, interval_tier.IntervalTier):
401                intervalTierNames.append(tierName)
402            elif isinstance(tier, point_tier.PointTier):
403                pointTierNames.append(tierName)
404
405        # Merge the interval tiers
406        intervalTier = None
407        if len(intervalTierNames) > 0:
408            intervalTier = self.getTier(intervalTierNames.pop(0))
409            for tierName in intervalTierNames:
410                intervalTier = intervalTier.union(self.getTier(tierName))
411
412        # Merge the point tiers
413        pointTier = None
414        if len(pointTierNames) > 0:
415            pointTier = self.getTier(pointTierNames.pop(0))
416            for tierName in pointTierNames:
417                pointTier = pointTier.union(self.getTier(tierName))
418
419        # Create the final textgrid to output
420        tg = Textgrid(self.minTimestamp, self.maxTimestamp)
421
422        if preserveOtherTiers:
423            for tier in self.tiers:
424                if tier.name not in tierNames:
425                    tg.addTier(tier)
426
427        if intervalTier is not None:
428            tg.addTier(intervalTier)
429
430        if pointTier is not None:
431            tg.addTier(pointTier)
432
433        return tg

440        self,
441        fn: str,
442        format: Literal["short_textgrid", "long_textgrid", "json", "textgrid_json"],
443        includeBlankSpaces: bool,
444        minTimestamp: Optional[float] = None,

466                if None, use whatever is defined in the Textgrid object.
467                If maxTimestamp is smaller than timestamps in your textgrid,

472                Removed intervals are merged (without their label) into
473                adjacent entries.

477
478        Returns:
479            a string representation of the textgrid
480        """
481
482        utils.validateOption("format", format, TextgridFormats)
483        utils.validateOption(
484            "reportingMode", reportingMode, constants.ErrorReportingMode
485        )
486
487        self.validate(reportingMode)
488
489        tgAsDict = _tgToDictionary(self)
490
491        textgridStr = textgrid_io.getTextgridAsStr(
492            tgAsDict,
493            format,
494            includeBlankSpaces,
495            minTimestamp,
496            maxTimestamp,
497            minimumIntervalLength,
498        )
499

522    def validate(
523        self, reportingMode: Literal["silence", "warning", "error"] = "warning"
524    ) -> bool:
525        """Validates this textgrid
526
527        Returns whether the textgrid is valid or not. If reportingMode is "warning"
528        or "error" this will also print on error or stop execution, respectively.
529
530        Args:
531            reportingMode: one of "silence", "warning", or "error". This flag
532                determines the behavior if there is a size difference between the
533                maxTimestamp in a tier and the current textgrid.
534
535        Returns:
536            True if this Textgrid is valid; False if not
537
538        Raises:
539            TierNameExistsError: Two tiers have the same name
540            TextgridStateError: A timestamp fall outside of the allowable range
541        """
542        utils.validateOption(
543            "reportingMode", reportingMode, constants.ErrorReportingMode
544        )
545        errorReporter = utils.getErrorReporter(reportingMode)
546
547        isValid = True
548        if len(self.tierNames) != len(set(self.tierNames)):
549            isValid = False
550            errorReporter(
551                errors.TierNameExistsError,
552                f"Tier names not unique: {self.tierNames}",
553            )
554
555        for tier in self.tiers:
556            if self.minTimestamp != tier.minTimestamp:
557                isValid = False
558                errorReporter(
559                    errors.TextgridStateError,
560                    f"Textgrid has a min timestamp of ({self.minTimestamp}) "
561                    f"but tier has ({tier.minTimestamp})",
562                )
563
564            if self.maxTimestamp != tier.maxTimestamp:
565                isValid = False
566                errorReporter(
567                    errors.TextgridStateError,
568                    f"Textgrid has a max timestamp of ({self.maxTimestamp}) "
569                    f"but tier has ({tier.maxTimestamp})",
570                )
571
572            isValid = isValid and tier.validate(reportingMode)
573
574        return isValid
